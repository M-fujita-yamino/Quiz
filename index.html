async function init() { try { document.getElementById('status').innerText = "通信開始..."; const response = await fetch(CSV_URL); const csvText = await response.text(); // 1. 改行コードの揺れ（\r\n または \n）を考慮して分割 const rows = csvText.split(/\r?\n/).filter(row => row.trim() !== ""); celebrities = rows.map((row, index) => { // 2. カンマで分割（引用符などは後でまとめて掃除する） const cols = row.split(','); // 3. 各列のゴミ（引用符、空白）を徹底的に掃除 const clean = (s) => s ? s.replace(/^["']|["']$/g, '').trim() : ""; const name = clean(cols[0]); const keywordsRaw = clean(cols[1]); const imageUrl = clean(cols[cols.length - 1]); // 一番右側の列をURLとみなす return { name: name, keywords: keywordsRaw ? keywordsRaw.split('|').map(k => k.trim()) : [], img: imageUrl }; }).filter(c => c.name && c.img && c.img.startsWith('http')); console.log("読み込んだデータ:", celebrities); // デバッグ用 if (celebrities.length > 0) { document.getElementById('status').innerText = "準備完了 (" + celebrities.length + "件)"; nextQuiz(); } else { document.getElementById('status').innerText = "有効なデータが1件もありません"; } } catch (e) { document.getElementById('status').innerText = "読込エラー: " + e.message; } }
